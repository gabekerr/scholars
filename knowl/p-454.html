\begin{equation*} \mb{v} = \textnormal{proj}_U (\mb{v} ) + \textnormal{proj}_{U^\perp} (\mb{v}) \end{equation*}

\begin{equation} \textnormal{proj}_U (\mb{v}) = \left\langle \mb{v}, \mb{v}_1 \right\rangle \mb{v}_1 + \cdots + \left\langle \mb{v} , \mb{v}_k \right\rangle \mb{v}_k. \tag{3.2.3} \end{equation}
